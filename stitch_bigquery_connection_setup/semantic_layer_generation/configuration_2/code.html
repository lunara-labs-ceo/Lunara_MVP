SELECT
  date_trunc(o.order_date, MONTH) AS month,
  p.category,
  sum(o.amount) AS total_revenue
FROM
  `lunara.public.orders` AS o
JOIN
  `lunara.public.products` AS p ON o.product_id = p.id
WHERE
  o.order_date >= '2023-01-01'
  AND p.category != 'Accessories'
GROUP BY
  1, 2
ORDER BY
  1 DESC, 2
